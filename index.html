0. 整体目标

默认路径不变不传 eagle_model 时继续使用原始 nano-vLLM decode。
EAGLE 显式启用只有单序列、greedy、非 prefill 阶段进入 EAGLE。
一次验证多 tokendraft 生成候选，target model 负责最终验证。
KV 状态安全预留 lookahead block，并回收未使用的 block。

1. EAGLE v1 工作流程

Input Prompt用户输入
→
Target Prefill写入 KV cache，产生 feature
→
EAGLE Draft预测 t1, t2, t3, t4
→
Target Verify一次验证多个候选 token
→
Accept Tokens提交连续匹配 token
→
Next Round更新 sequence 和 KV cache

EAGLE 不直接决定最终输出。最终 token 仍由 target model 验证决定，所以 greedy 场景可以保持和 vanilla decoding 一致。

2. 原始 nano-vLLM 与 EAGLE Decode 对比

Vanilla nano-vLLM Decode
Target Step 1token A
Target Step 2token B
Target Step 3token C
Target Step 4token D

特点：每生成一个 token，都要调用一次 target model decode。

nano-vLLM + EAGLE Decode
Target Featuresecond-top-layer feature
Draft TokensA, B, C, D
Verify Oncetarget logits
Accept Togetherappend multiple tokens

特点：target model 一次验证多个 draft token，减少 decode 轮数。

class SamplingParams:
    temperature: float = 1.0
    max_tokens: int = 64
    ignore_eos: bool = False

    def __post_init__(self):
        assert self.temperature > 1e-10, \
            "greedy sampling is not permitted"


class Sampler(nn.Module):

    @torch.compile
    def forward(self, logits, temperatures):
        logits = logits.float().div_(temperatures.unsqueeze(dim=1))
        probs = torch.softmax(logits, dim=-1)
        sample_tokens = probs.div_(
            torch.empty_like(probs)
            .exponential_(1)
            .clamp_min_(1e-10)
        ).argmax(dim=-1)
        return sample_tokens

class Sampler(nn.Module):

    def forward(self, logits, temperatures):
        if torch.all(temperatures <= 1e-10):
            return logits.argmax(dim=-1)

        logits = logits.float().div_(temperatures.unsqueeze(dim=1))
        probs = torch.softmax(logits, dim=-1)
        sample_tokens = probs.div_(
            torch.empty_like(probs)
            .exponential_(1)
            .clamp_min_(1e-10)
        ).argmax(dim=-1)
        return sample_tokens

5. Qwen3Model：暴露 EAGLE Feature

原始 nano-vLLM
class Qwen3Model(nn.Module):

    def forward(self, input_ids, positions):
        hidden_states = self.embed_tokens(input_ids)
        residual = None

        for layer in self.layers:
            hidden_states, residual = layer(
                positions,
                hidden_states,
                residual,
            )

        hidden_states, _ = self.norm(hidden_states, residual)
        return hidden_states
接入 EAGLE 后
class Qwen3Model(nn.Module):

    def forward(
        self,
        input_ids,
        positions,
        return_eagle_features: bool = False,
    ):
        hidden_states = self.embed_tokens(input_ids)
        residual = None
        eagle_features = None

        for i, layer in enumerate(self.layers):
            if return_eagle_features and i == len(self.layers) - 1:
                eagle_features = (
                    hidden_states if residual is None
                    else hidden_states + residual
                )
            hidden_states, residual = layer(
                positions,
                hidden_states,
                residual,
            )

        hidden_states, _ = self.norm(hidden_states, residual)
        if return_eagle_features:
            return hidden_states, eagle_features
        return hidden_states

class EagleDraftModel(nn.Module):

    def __init__(self, config):
        super().__init__()
        self.embed_tokens = nn.Embedding(
            config.vocab_size,
            config.hidden_size,
            getattr(config, "pad_token_id", 0),
        )
        self.fc = nn.Linear(
            2 * config.hidden_size,
            config.hidden_size,
            bias=getattr(config, "bias", True),
        )
        self.layers = nn.ModuleList([
            EagleDecoderLayer(config)
            for _ in range(config.num_hidden_layers)
        ])

    def forward(self, features, input_ids, position_offset=0):
        inputs_embeds = self.embed_tokens(input_ids).to(features.dtype)
        x = self.fc(torch.cat((inputs_embeds, features), dim=-1))
        positions = torch.arange(
            position_offset,
            position_offset + input_ids.size(1),
            device=input_ids.device,
        )
        for layer in self.layers:
            x = layer(x, positions)
        return x

真实加速需要训练好的 EAGLE draft 权重。只有 config 的 dummy draft 可以验证流程，但不能代表真实性能。

def run_eagle(self, seqs):
    seq = seqs[0]

    input_ids, positions = self.prepare_decode(seqs)
    logits, features = self.run_model_with_features(input_ids, positions)
    seed_token = int(torch.argmax(logits[-1], dim=-1).item())
    seed_feature = features[-1]
    reset_context()

    candidates = self.generate_eagle_candidates(
        seq,
        seed_token,
        seed_feature,
    )

    verify_ids = [seq.last_token] + candidates
    input_ids, positions = self.prepare_extend(seq, verify_ids)
    logits, features = self.run_model_with_features(input_ids, positions)
    reset_context()

    accepted = []
    for i, candidate in enumerate(candidates):
        target_token = int(torch.argmax(logits[i], dim=-1).item())
        if target_token == candidate:
            accepted.append(candidate)
        else:
            accepted.append(target_token)
            return accepted

    accepted.append(int(torch.argmax(logits[len(candidates)], dim=-1).item()))
    return accepted

def schedule(self, lookahead_slots: int = 1):
    ...
    while self.running and num_seqs < self.max_num_seqs:
        seq = self.running.popleft()
        while not self.block_manager.can_append_slots(seq, lookahead_slots):
            ...
        else:
            num_seqs += 1
            if lookahead_slots == 1:
                self.block_manager.may_append(seq)
            else:
                self.block_manager.ensure_append_slots(seq, lookahead_slots)
            scheduled_seqs.append(seq)


def postprocess_many(self, seq, token_ids):
    old_len = len(seq)
    for token_id in token_ids:
        if seq.num_completion_tokens >= seq.max_tokens:
            break
        seq.append_token(token_id)
        if not seq.ignore_eos and token_id == self.eos:
            break

    self.block_manager.trim_unused_blocks(seq)
    self.block_manager.commit_appended_tokens(seq, old_len)

    if should_finish(seq):
        seq.status = SequenceStatus.FINISHED
        self.block_manager.deallocate(seq)
        self.running.remove(seq)

9. LLMEngine：选择 vanilla path 或 EAGLE path

原始 nano-vLLM
def step(self):
    seqs, is_prefill = self.scheduler.schedule()
    token_ids = self.model_runner.call("run", seqs, is_prefill)
    self.scheduler.postprocess(seqs, token_ids)

    outputs = [
        (seq.seq_id, seq.completion_token_ids)
        for seq in seqs
        if seq.is_finished
    ]
    num_tokens = (
        sum(len(seq) for seq in seqs)
        if is_prefill
        else -len(seqs)
    )
    return outputs, num_tokens
接入 EAGLE 后
def step(self):
    can_try_eagle = (
        self.config.eagle_model is not None
        and not self.scheduler.waiting
        and len(self.scheduler.running) == 1
    )
    lookahead_slots = (
        self.config.eagle_max_draft_tokens + 1
        if can_try_eagle
        else 1
    )
    seqs, is_prefill = self.scheduler.schedule(lookahead_slots)

    if (
        self.config.eagle_model is not None
        and not is_prefill
        and len(seqs) == 1
        and seqs[0].temperature <= 1e-10
    ):
        token_ids = self.model_runner.call("run_eagle", seqs)
        self.scheduler.postprocess_many(seqs[0], token_ids)
    else:
        token_ids = self.model_runner.call("run", seqs, is_prefill)
        self.scheduler.postprocess(seqs, token_ids)

    outputs = [
        (seq.seq_id, seq.completion_token_ids)
        for seq in seqs
        if seq.is_finished
    ]
    num_tokens = (
        sum(len(seq) for seq in seqs)
        if is_prefill
        else -len(token_ids)
    )
    return outputs, num_tokens

11. 总结

在 nano-vLLM 上部署 EAGLE，关键不是简单地增加一个小模型，而是把 speculative decoding 的状态流接进推理系统： target model 暴露 feature，draft model 生成候选，target model 一次验证多个 token，Scheduler 一次提交多个 token， BlockManager 负责 lookahead KV block 的预留和回收。

这个版本保持默认路径不变，只在单序列 greedy decode 下显式进入 EAGLE 路径，适合作为继续扩展 tree draft、 非 greedy speculative sampling 和 batch EAGLE 的基础。
